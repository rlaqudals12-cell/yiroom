    }

    nutrientScores[nutrient] = Math.round(Math.max(0, Math.min(100, score)));
    totalScore += nutrientScores[nutrient];
    validCount++;
  }

  const overallScore = validCount > 0 ? Math.round(totalScore / validCount) : 0;

  return {

    nutrientScores,
    deficiencies,
