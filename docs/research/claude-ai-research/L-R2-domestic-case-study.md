### 2.3 체형 기반 추천 시스템

```typescript
// 룩핀 체형 매칭 (추정 로직)
interface UserBodyProfile {
  height: number;       // cm
  weight: number;       // kg
  shoulderWidth: number; // cm
  waistSize: number;    // cm
  bodyType: 'slim' | 'standard' | 'athletic' | 'curvy';
}

function findSimilarUsers(profile: UserBodyProfile): User[] {
  // 유클리드 거리 기반 유사 사용자 검색
  const normalizedProfile = normalize(profile);

  return users
    .map(user => ({
      user,
      distance: euclideanDistance(normalizedProfile, normalize(user.profile))
    }))
    .sort((a, b) => a.distance - b.distance)
    .slice(0, 50)
    .map(item => item.user);
}

function getRecommendedOutfits(userId: string): Outfit[] {
  const profile = getProfile(userId);
  const similarUsers = findSimilarUsers(profile);

  // 유사 사용자의 착용샷에서 추천
  return similarUsers
    .flatMap(user => user.outfits)
    .filter(outfit => outfit.rating >= 4.0)
    .sort((a, b) => b.engagement - a.engagement);
}
```

3. **브랜드 가치 증명**: 올리브영처럼 브랜드에게 데이터 가치 제공
